const username = document.getElementById('username').value; const password = document.getElementById('password').value; // Define los usuarios y contraseñas válidos const validUsers = { 'tmvinfo': '1234', 'cliente': 'cliente' }; // Comprueba si el usuario existe y la contraseña es correcta if (validUsers[username] && validUsers[username] === password) { // Si es correcto, oculta el formulario y muestra el menú document.getElementById('login-container').style.display = 'none'; document.getElementById('main-menu').style.display = 'block'; } else { // Si no es correcto, muestra un mensaje de error alert('Usuario o contraseña incorrectos.'); } }); document.getElementById('logout-button').addEventListener('click', function() { // Limpia los campos de texto document.getElementById('username').value = ''; document.getElementById('password').value = ''; // Oculta el menú y muestra de nuevo el formulario de login document.getElementById('main-menu').style.display = 'none'; document.getElementById('login-container').style.display = 'block'; });
